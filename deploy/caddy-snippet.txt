# Add these three lines inside the EXISTING intranet site block in
# /opt/caddy/Caddyfile so it becomes:
#
# intranet.groundlinkhandling.pt {
#     redir /dashboards/safety /dashboards/safety/
#
#     handle_path /dashboards/safety/* {
#         reverse_proxy safety:3000
#     }
#
#     handle_path /api/* {
#         reverse_proxy backend:4000
#     }
#     handle {
#         reverse_proxy frontend:80
#     }
# }
#
# Notes:
# - handle_path strips the /dashboards/safety prefix, so the dashboard receives
#   /, /logo.png and /api/* — exactly what its server.js serves.
# - redir sends the no-trailing-slash visit to /dashboards/safety/ so the app's
#   <base href> resolves correctly.
# - Caddy orders handle blocks by matcher specificity, so this coexists with the
#   existing /api/* and catch-all routes regardless of where you paste it.
#
# Reload Caddy after editing (no downtime):
#   docker exec caddy-caddy-1 caddy reload --config /etc/caddy/Caddyfile
